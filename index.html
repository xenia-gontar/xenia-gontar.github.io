🎯 Воронка продаж
Сколько сделок на каждой стадии и их общая сумма
SELECT
    CASE STAGE_SEMANTIC_ID
        WHEN 'P' THEN 'В работе'
        WHEN 'S' THEN 'Успешно'
        WHEN 'F' THEN 'Проиграно'
    END AS status,
    COUNT(*) AS deals_count,
    ROUND(SUM(OPPORTUNITY), 0) AS total_amount
FROM crm_deal
GROUP BY STAGE_SEMANTIC_ID;
📐 Использованные конструкции:
CASE WHEN — преобразование кодов в читаемые названия
COUNT(*) — подсчёт количества записей
SUM() — агрегация суммы сделок
ROUND() — округление чисел
GROUP BY — группировка по стадиям

👥 Рейтинг менеджеров по выручке
Кто из ответственных принёс больше всего денег
SELECT
    ASSIGNED_BY_NAME AS manager,
    COUNT(*) AS deals_count,
    ROUND(SUM(CASE WHEN STAGE_SEMANTIC_ID = 'S' THEN OPPORTUNITY ELSE 0 END), 0) AS revenue
FROM crm_deal
GROUP BY ASSIGNED_BY_NAME
ORDER BY revenue DESC;
📐 Использованные конструкции:
CASE WHEN в SUM() — условная агрегация (считаем только успешные сделки)
ELSE 0 — защита от NULL в сумме
ORDER BY DESC — сортировка по убыванию для рейтинга
AS — понятные алиасы для вывода

📈 Динамика продаж по месяцам
Новые сделки и выручка помесячно
SELECT
    TO_CHAR(DATE_CREATE, 'YYYY-MM') AS month,
    COUNT(*) AS new_deals,
    ROUND(SUM(CASE WHEN STAGE_SEMANTIC_ID = 'S' THEN OPPORTUNITY ELSE 0 END), 0) AS revenue
FROM crm_deal
GROUP BY TO_CHAR(DATE_CREATE, 'YYYY-MM')
ORDER BY month;
📐 Использованные конструкции:
TO_CHAR() — функция преобразования даты в строку (формат ГГГГ-ММ)
GROUP BY с функцией — группировка по результату функции
CASE WHEN в SUM() — условная агрегация
ORDER BY month — хронологическая сортировка

🔮 Прогноз на следующий месяц
Сделки в работе и ожидаемая выручка с учётом вероятности
WITH weighted_deals AS (
    SELECT
        OPPORTUNITY,
        PROBABILITY / 100.0 AS weight
    FROM crm_deal
    WHERE STAGE_SEMANTIC_ID = 'P'
)
SELECT
    'Прогноз на апрель 2025' AS period,
    COUNT(*) AS pipeline_deals,
    ROUND(AVG(OPPORTUNITY), 0) AS avg_deal,
    SUM(OPPORTUNITY * weight) AS expected_revenue
FROM weighted_deals;
📐 Использованные конструкции:
WITH (CTE) — обобщённое табличное выражение для читаемости
AVG() — среднее арифметическое
SUM() с весами — взвешенная сумма (вероятность × сумма сделки)
/ 100.0 — преобразование процентов в доли

📉 Анализ проигранных сделок
На каких стадиях чаще всего теряем сделки
SELECT
    stage_name,
    COUNT(*) AS lose_count,
    ROUND(AVG(avg_days_in_stage), 1) AS avg_days
FROM (
    SELECT
        STAGE_NAME,
        DAYS_IN_STAGE AS avg_days_in_stage
    FROM crm_deal_stage_history
    WHERE DEAL_ID IN (
        SELECT DEAL_ID FROM crm_deal WHERE STAGE_SEMANTIC_ID = 'F'
    )
    ) AS lose_stages
GROUP BY stage_name
ORDER BY lose_count DESC;
📐 Использованные конструкции:
Подзапрос в FROM — вложенный SELECT как источник данных
IN (подзапрос) — фильтрация по результату другого запроса
AVG() с подзапросом — среднее по отфильтрованным данным
Двойная вложенность — подзапрос в подзапросе

🔄 Конверсия между стадиями
Сколько сделок переходит со стадии на стадию
WITH stage_transitions AS (
    SELECT
        DEAL_ID,
        STAGE_NAME,
        LAG(STAGE_NAME) OVER (PARTITION BY DEAL_ID ORDER BY DATE_CREATE) AS prev_stage
    FROM crm_deal_stage_history
)
SELECT
    prev_stage AS from_stage,
    STAGE_NAME AS to_stage,
    COUNT(*) AS transitions
FROM stage_transitions
WHERE prev_stage IS NOT NULL
GROUP BY prev_stage, STAGE_NAME
ORDER BY transitions DESC;
📐 Использованные конструкции:
LAG() OVER (PARTITION BY... ORDER BY...) — оконная функция для доступа к предыдущей строке в группе
PARTITION BY DEAL_ID — сброс окна для каждой сделки отдельно
ORDER BY DATE_CREATE — правильный порядок стадий во времени
WITH (CTE) — для читаемости сложной логики
